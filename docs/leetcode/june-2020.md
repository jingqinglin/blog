public static int[] dailyTemperatures(int[] T) {
    int len = T.length;
    int[] res = new int[len];
    LinkedList<Integer> stack = new LinkedList<>();
    for (int i = len - 1; i >= 0; i--) {

        stack.addLast(i);
    }
    return res;
}
